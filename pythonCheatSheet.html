def binary_search(arr, target):
    l, r = 0, len(arr)-1
    while l <= r:
        mid = (l+r)//2
        if   arr[mid] == target: return mid
        elif arr[mid] < target: l = mid+1
        else:                    r = mid-1
    return -1

BFS & DFS
from collections import deque

# BFS: shortest path, levels
def bfs(start, graph):
    visited = {start}
    q = deque([start])
    while q:
        node = q.popleft()
        for nb in graph[node]:
            if nb not in visited:
                visited.add(nb); q.append(nb)

# DFS: explore deep, detect cycles
def dfs(node, visited=None):
    visited = visited or set()
    visited.add(node)
    for nb in graph.get(node,[]):
        if nb not in visited: dfs(nb,visited)

Hard Problem 1 — LRU Cache (OrderedDict)
from collections import OrderedDict

class LRUCache:
    def __init__(self, capacity):
        self.cap = capacity
        self.cache = OrderedDict()

    def get(self, key):
        if key not in self.cache: return -1
        self.cache.move_to_end(key)   # mark as recently used
        return self.cache[key]

    def put(self, key, value):
        self.cache[key] = value
        self.cache.move_to_end(key)
        if len(self.cache) > self.cap:
            self.cache.popitem(last=False)  # evict LRU
Hard Problem 2 — Longest Substring Without Repeating Characters
def length_of_longest_substring(s):
    seen = {}
    l = best = 0
    for r, c in enumerate(s):
        if c in seen:
            l = max(l, seen[c] + 1)  # jump left pointer past duplicate
        seen[c] = r
        best = max(best, r - l + 1)
    return best  # O(n) time, O(k) space
Hard Problem 3 — Word Break (DP on String)
def word_break(s, wordDict):
    word_set = set(wordDict)            # O(1) lookup
    n = len(s)
    dp = [False] * (n + 1)
    dp[0] = True                         # empty string is always segmentable
    for i in range(1, n + 1):
        for j in range(i):
            if dp[j] and s[j:i] in word_set:
                dp[i] = True
                break
    return dp[n]  # O(n²) time
Hard Problem 4 — Number of Islands (DFS on Grid)
def num_islands(grid):
    if not grid: return 0
    islands = 0
    def dfs(r, c):
        if r<0 or r>=len(grid) or c<0 or c>=len(grid[0]) or grid[r][c]!="1":
            return
        grid[r][c] = "0"  # flood fill visited
        dfs(r+1,c); dfs(r-1,c); dfs(r,c+1); dfs(r,c-1)
    for r in range(len(grid)):
        for c in range(len(grid[0])):
            if grid[r][c] == "1":
                islands += 1; dfs(r, c)
    return islands

What Is Tkinter? — Your First GUI App

Tkinter is Python's built-in library for building desktop windows and apps.

Think of it like building with LEGO bricks — each piece (button, label, input box) is called a widget. You place widgets inside a window, and tell Python what to do when someone clicks or types.

Best part: it comes with Python — nothing to install!

  • Already installed — just import tkinter as tk
  • Works on Windows, Mac, and Linux
  • Perfect for calculators, todo apps, simple games
  • Step 1: create window → Step 2: add widgets → Step 3: start loop
The 3 Steps — Every Tkinter App Uses This

import tkinter as tk

root = tk.Tk()
root.title("Calculator")
root.resizable(False, False)

display = tk.StringVar(value="0")
tk.Entry(root, textvariable=display,
         font=("Arial",24), justify="right",
         state="readonly", bd=0).grid(row=0, column=0,
                                              columnspan=4, sticky="ew", padx=4, pady=4)

def press(key):
    cur = display.get()
    if key == "=":
        try: display.set(eval(cur))
        except: display.set("Error")
    elif key == "C": display.set("0")
    else: display.set("" if cur=="0" else cur).__class__ or display.set(cur+key) if 1 else None

def press2(key):  # cleaner version
    cur = display.get()
    if   key == "=":
        try: display.set(eval(cur))
        except: display.set("Error")
    elif key == "C": display.set("0")
    elif cur == "0": display.set(key)
    else: display.set(cur + key)

buttons = [
    ["7","8","9","/"],
    ["4","5","6","*"],
    ["1","2","3","-"],
    ["C","0","=","+"],
]
for r, row in enumerate(buttons, 1):
    for c, key in enumerate(row):
        bg = "#f5c842" if key in "=C" else "#252535"
        tk.Button(root, text=key, font=("Arial",18),
                  bg=bg, fg="black" if bg=="#f5c842" else "white",
                  width=4, height=2,
                  command=lambda k=key: press2(k)).grid(
            row=r, column=c, padx=2, pady=2)

⚠️ Common Beginner Mistakes — Read This!

Wrong ✗

# 1. Calling the function instead of referencing it
tk.Button(root, command=my_func())  # BAD: () runs it now!

# 2. Forgetting to pack/grid/place
tk.Label(root, text="Hi")  # invisible! no .pack()

# 3. mainloop() in the middle
root.mainloop()
tk.Label(root, text="never reached").pack()

# 4. Lambda captures variable by ref (loop bug)
for i in range(3):
    tk.Button(root, command=lambda: print(i)).pack()
    # all buttons print 2! (last value of i)

Correct ✓

# 1. Reference the function (no parentheses!)
tk.Button(root, command=my_func)   # GOOD
tk.Button(root, command=lambda: f(arg))  # with args

# 2. Always pack/grid/place after creating
tk.Label(root, text="Hi").pack()   # GOOD

# 3. mainloop() is ALWAYS the last line
# add all widgets THEN call mainloop()
root.mainloop()  # ← last line of your file

# 4. Capture loop var with default argument
for i in range(3):
    tk.Button(root, command=lambda x=i: print(x)).pack()
    # each button prints correct number ✓

The "Self" & Real-World OOP

Self: When you call user1.login(), Python passes user1 as the first argument automatically. It's how the method knows *which* object it's working on.

class VocaUser:
    def __init__(self, username):
        self.username = username
        self.is_logged_in = False

    def login(self):
        self.is_logged_in = True
        print(f"{self.username} entered!")

# Usage
u1 = VocaUser("Kero")
u1.login()  # self becomes u1

💡 Mini-Challenge: Frequency Count

Problem: Count how many times each word appears in a list.

    counts[w] = counts.get(w, 0) + 1

Error Handling (Try/Except)

Prevent your app from crashing when things go wrong.

try:
    num = int(input("Enter number: "))
    result = 10 / num
except ValueError:
    print("That's not a number!")
except ZeroDivisionError:
    print("You can't divide by zero!")
finally:
    print("Cleanup complete.")
